\section{Formula Bank}
\label{sec:formula_bank}

\subsection{Notation}
\label{subsec:formula_notation}

Let $x$ denote an input image. AnomalyCLIP extracts an image feature
$q \in \mathbb{R}^{C}$ and patch features
$Z^{\ell}=\{z_i^{\ell}\}_{i=1}^{N}$ from layer $\ell$, where
$z_i^{\ell}\in\mathbb{R}^{C}$. The normal and abnormal text prototypes are
$t_n,t_a\in\mathbb{R}^{C}$. We use $\mathcal{N}(\cdot)$ for per-image
min-max normalization to $[0,1]$, and $\langle \cdot,\cdot\rangle$ for cosine
similarity after feature normalization.

\subsection{AnomalyCLIP Patch Scoring}
\label{subsec:formula_clip_scoring}

For each selected layer $\ell$, the normal and abnormal patch similarities are
\begin{equation}
  s_{i,n}^{\ell} = \langle z_i^{\ell}, t_n \rangle,
  \qquad
  s_{i,a}^{\ell} = \langle z_i^{\ell}, t_a \rangle .
\end{equation}

The layer-wise patch anomaly response is
\begin{equation}
  A_i^{\ell}
  =
  \frac{s_{i,a}^{\ell} + 1 - s_{i,n}^{\ell}}{2}.
\end{equation}

After reshaping patch scores to the image grid and resizing to the image
resolution, the baseline anomaly map is fused across selected layers
$\mathcal{L}$ by
\begin{equation}
  A_0(p) = \sum_{\ell\in\mathcal{L}} A^{\ell}(p).
\end{equation}

The image-level anomaly probability is computed from the CLIP image feature:
\begin{equation}
  s_{\mathrm{img}}
  =
  \frac{\exp(\langle q,t_a\rangle/\tau)}
       {\exp(\langle q,t_n\rangle/\tau)+\exp(\langle q,t_a\rangle/\tau)} .
\end{equation}

\subsection{Feature-Space Haar Wavelet Decomposition}
\label{subsec:formula_wavelet_decomposition}

For a $2\times2$ neighborhood in the patch-feature grid,
$X_{00},X_{01},X_{10},X_{11}\in\mathbb{R}^{C}$, the Haar components are
\begin{align}
  LL &= \frac{1}{2}(X_{00}+X_{01}+X_{10}+X_{11}), \\
  LH &= \frac{1}{2}(X_{00}-X_{01}+X_{10}-X_{11}), \\
  HL &= \frac{1}{2}(X_{00}+X_{01}-X_{10}-X_{11}), \\
  HH &= \frac{1}{2}(X_{00}-X_{01}-X_{10}+X_{11}).
\end{align}

The high-frequency energy at a wavelet level is
\begin{equation}
  E =
  \sqrt{
    \frac{1}{C}
    \sum_{c=1}^{C}
    \left( LH_c^2 + HL_c^2 + HH_c^2 \right)
    + \epsilon
  } .
\end{equation}

For multi-level decomposition, the next level is recursively computed from
the low-frequency component $LL$. The resized and normalized high-frequency
gate for layer $\ell$ is
\begin{equation}
  W^{\ell}
  =
  \mathcal{N}
  \left(
    \frac{1}{M}\sum_{m=1}^{M}
    \operatorname{Up}\left(\mathcal{N}(E_m^{\ell})\right)
  \right),
\end{equation}
where $M$ is the number of wavelet levels. The final wavelet gate is fused
across layers:
\begin{equation}
  W = \mathcal{N}
  \left(
    \frac{1}{|\mathcal{L}|}
    \sum_{\ell\in\mathcal{L}} W^{\ell}
  \right).
\end{equation}

\subsection{Structure-Texture Separation}
\label{subsec:formula_structure_texture}

The low-frequency structure edge is estimated from gradients of the
low-frequency feature grid:
\begin{equation}
  S =
  \mathcal{N}
  \left(
    \sqrt{
      \left\|\nabla_x LL\right\|_2^2
      +
      \left\|\nabla_y LL\right\|_2^2
      + \epsilon
    }
  \right).
\end{equation}

The texture gate suppresses high-frequency responses that coincide with
semantic structure edges:
\begin{equation}
  T = \mathcal{N}\left(W \odot (1-S)^{\gamma}\right),
\end{equation}
where $\gamma$ is the structure-edge suppression power.

\subsection{Wavelet Reliability}
\label{subsec:formula_reliability}

Let $P=\mathcal{N}(A_0)$ be the normalized CLIP anomaly prior. The global
agreement between CLIP and wavelet evidence combines positive correlation and
top-response overlap. First,
\begin{equation}
  \rho(P,T)
  =
  \max
  \left(
    0,
    \frac{
      \sum_p (P_p-\bar{P})(T_p-\bar{T})
    }{
      \sqrt{\sum_p(P_p-\bar{P})^2}
      \sqrt{\sum_p(T_p-\bar{T})^2}
      + \epsilon
    }
  \right).
\end{equation}

Given top-$k$ pixel sets $\Omega_P^k$ and $\Omega_T^k$, the overlap score is
\begin{equation}
  o(P,T)
  =
  \left[
    \frac{
      |\Omega_P^k \cap \Omega_T^k|/k - r
    }{
      1-r+\epsilon
    }
  \right]_{0}^{1},
\end{equation}
where $r=k/|\Omega|$ is the random-overlap baseline and $[\cdot]_0^1$ clips
values to $[0,1]$. The image-wise reliability is
\begin{equation}
  R = \sqrt{\rho(P,T)\,o(P,T)+\epsilon}.
\end{equation}

\subsection{Wavelet-Guided Map Calibration}
\label{subsec:formula_wavelet_calibration}

The texture evidence is used only where it is semantically compatible with the
CLIP prior. With rank gate $M_k(P)$ from the top-$k$ CLIP anomaly pixels, the
semantic texture evidence is
\begin{equation}
  U = T \odot P^{\eta} \odot M_k(P),
\end{equation}
where $\eta$ is the CLIP-conditioning power.

The local CLIP-wavelet confidence is
\begin{equation}
  C_{\mathrm{local}} = \left(1-|P-T|\right)^{\mu}.
\end{equation}

The final confidence gate used by the current main configuration is
\begin{equation}
  C = R^{\lambda} \odot C_{\mathrm{local}}.
\end{equation}

Let $\Delta_A=\max_p A_0(p)-\min_p A_0(p)$ be the image-wise anomaly-score
range. The bounded wavelet bonus is
\begin{equation}
  B(p)
  =
  \min
  \left(
    \beta\, C(p)\, U(p)\,\Delta_A,
    \delta_{\max}\Delta_A
  \right).
\end{equation}

The calibrated anomaly map is
\begin{equation}
  A_{\mathrm{wav}}(p) = A_0(p) + B(p).
\end{equation}

\subsection{Wavelet-Guided Test-Time Adaptation}
\label{subsec:formula_tta}

TTA builds normal and abnormal visual anchors from reliable patches without
labels. Patch confidence scores are
\begin{equation}
  c_i^n = (1-P_i)(1-T_i),
  \qquad
  c_i^a = P_iT_i .
\end{equation}

For a confidence vector $c$, the top-$k$ weighted visual anchor is
\begin{equation}
  v(c)
  =
  \operatorname{Norm}
  \left(
    \sum_{i\in\Omega_c^k}
    \frac{c_i}{\sum_{j\in\Omega_c^k} c_j+\epsilon}
    z_i
  \right).
\end{equation}

Thus,
\begin{equation}
  v_n = v(c^n),
  \qquad
  v_a = v(c^a).
\end{equation}

In wavelet-guided mode, the normal anchor is repelled from abnormal evidence:
\begin{equation}
  \tilde{v}_n
  =
  \operatorname{Norm}(v_n-\xi v_a),
\end{equation}
where $\xi$ is the repulsion weight.

With effective update strength $\alpha_{\mathrm{eff}}=\alpha R$, the text
prototypes are rectified at test time by
\begin{align}
  \tilde{t}_n
  &=
  \operatorname{Norm}
  \left(
    (1-\alpha_{\mathrm{eff}})t_n
    +
    \alpha_{\mathrm{eff}}\tilde{v}_n
  \right), \\
  \tilde{t}_a
  &=
  \operatorname{Norm}
  \left(
    (1-\kappa\alpha_{\mathrm{eff}})t_a
    +
    \kappa\alpha_{\mathrm{eff}}v_a
  \right),
\end{align}
where $\kappa$ is the abnormal-anchor update scale.

For multi-layer TTA, the current implementation averages the per-layer
rectified text features:
\begin{equation}
  \tilde{t}
  =
  \operatorname{Norm}
  \left(
    \frac{1}{|\mathcal{L}|}
    \sum_{\ell\in\mathcal{L}} \tilde{t}^{\ell}
  \right).
\end{equation}

\subsection{Multi-Crop Localization Fusion}
\label{subsec:formula_multicrop}

For crop boxes $\mathcal{B}=\{b_j\}_{j=1}^{J}$, each crop produces an anomaly
map $A_j$. The crop maps are resized and pasted back to the original image
grid. For pixel $p$, the stitched multi-crop map is
\begin{equation}
  A_{\mathrm{mc}}(p)
  =
  \frac{
    \sum_{j=1}^{J}
    \mathbb{1}[p\in b_j]\,
    \operatorname{Paste}_{b_j}(A_j)(p)
  }{
    \sum_{j=1}^{J}\mathbb{1}[p\in b_j] + \epsilon
  }.
\end{equation}

The final pixel map fuses the full-image route and the multi-crop route:
\begin{equation}
  A_{\mathrm{fuse}}
  =
  (1-\omega)A_{\mathrm{wav}}
  +
  \omega A_{\mathrm{mc}},
\end{equation}
where the current best MVTec setting uses $\omega=0.50$.

\subsection{Pixel-to-Image Score Fusion}
\label{subsec:formula_pixel_to_image}

The pixel-level top-$k$ score is
\begin{equation}
  s_{\mathrm{pix}}
  =
  \frac{1}{k}
  \sum_{p\in\Omega_{A}^{k}}
  A_{\mathrm{fuse}}(p),
\end{equation}
where $\Omega_A^k$ contains the top-$k$ pixels of the fused anomaly map.

The final image-level score is
\begin{equation}
  s_{\mathrm{final}}
  =
  (1-\lambda)s_{\mathrm{img}}
  +
  \lambda s_{\mathrm{pix}}.
\end{equation}

\subsection{Gaussian Map Smoothing}
\label{subsec:formula_smoothing}

The reported pixel anomaly map is smoothed by a Gaussian kernel:
\begin{equation}
  A_{\mathrm{out}} = G_{\sigma} * A_{\mathrm{fuse}}.
\end{equation}
